users:
  ? national_id               (رقم الهوية الوطنية)
  ? national_id_front_url     (صورة أمام الهوية)
  ? national_id_back_url      (صورة خلف الهوية)
  ? license_card_url          (رخصة العمل)
  ? is_approved               (هل وافق عليه Admin؟)
  ? bio                       (نبذة عن الفني)
  ? years_of_experience       (سنوات الخبرة)
  ? rating / rating_count     (التقييم)
orders:
  ? scheduled_at              (وقت الطلب المجدوَل)
  ? lead_time_minutes         (وقت الاستعداد المطلوب)
  ? client_rating             (تقييم العميل للفني)
  ? tech_rating               (تقييم الفني للعميل)
  ? specialty_id              (مرتبط بـ service_specializations)
admins:
  ? role enum (super_admin/admin/operator)  ? حالياً: boolean فقط
الـ Backend — منطق ناقص:
  ? SELECT FOR UPDATE عند قبول الطلب (Race Condition)
  ? Dynamic Radius Expansion (توسع نطاق البحث تلقائياً)
  ? Heartbeat API لتتبع موقع الفني (خلال الطلب النشط)
  ? Presigned URLs للوثائق الخاصة (هوية الفني)
  ? Lead Time validation للطلبات المجدوَلة
  ? Technician Approval workflow
الـ Frontend — شاشات ناقصة:
  ? تسجيل الفني: رفع صور الهوية + رخصة العمل
  ? موافقة Admin على الفني الجديد
  ? تقييم الطلب (نجوم للعميل وللفني)
  ? الطلبات المجدوَلة (اختيار وقت مستقبلي)
  ? دور Operator في لوحة Admin

?? مقترحاتي الإضافية
1. لا تُنشئ جدول addresses منفصلاً الآن — جدول users يكفي. ضعه في المرحلة 3 فقط عندما يحتاج العميل عناوين متعددة.
2. استخدم jsonb للجدول الزمني للفني بدلاً من عمودَي service_start/service_end — يتيح تحديد أوقات مختلفة لكل يوم.
3. احتفظ بـ Nominatim كمصدر أساسي — أضف fallback لـ Google Places فقط إذا فشل (لا تدفع مقابله من البداية).
4. Race Condition — أولوية عالية، يجب حلها قبل الإطلاق الحقيقي.
